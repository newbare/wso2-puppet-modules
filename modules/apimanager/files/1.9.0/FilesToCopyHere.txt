How to add SSL certs to keystores
---------------------------------
keytool -import -file /etc/puppet/modules/nginx/files/ssl/server.crt -alias nginx -keystore wso2carbon.jks -storepass wso2carbon
keytool -import -file /etc/puppet/modules/nginx/files/ssl/server.crt -alias nginx -keystore client-truststore.jks -storepass wso2carbon
